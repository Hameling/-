#mysql 서버 확인
service mysql [명령어]
restart, status 등

#mysql 로그인
mysql -u[사용자 이름] -p
비밀번호 입력필요

#git 사용법
git status -> 현재 스테이징될 파일이 있는지 확인
git add [파일명] -> 해당 파일 스테이징
git add . -> 모든 파일 스테이징
git commit -m "메세지" -> 해당 메세지로 커밋
git push origin master -> 오리진 마스터로 푸쉬, 깃허브 아이디와 비밀번호 필요
